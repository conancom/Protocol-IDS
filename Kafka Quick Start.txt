# Start Kafka Environment

Terminal 1
sudo bin/zookeeper-server-start.sh config/zookeeper.properties

Terminal 2
sudo bin/kafka-server-start.sh config/server.properties

# Start Producer (write)

Terminal 3
sudo bin/kafka-console-producer.sh --topic get --bootstrap-server localhost:9092

# Start Consumer (read)

Terminal 4
sudo bin/kafka-console-consumer.sh --topic get --from-beginning --bootstrap-server localhost:9092


## Create Topic Event

sudo bin/kafka-topics.sh --create --topic get --bootstrap-server localhost:9092

## Delete Topic
sudo bin/kafka-topics.sh localhost:2181 --delete --topic get --bootstrap-server localhost:9092
